{
  "phase": "ASSESS",
  "q": 1,              // 当前题号
  "nb": 0,             // N(能动) 分数
  "bh": 0,             // B(边界) 分数
  "tf": 0,             // T(理性) 分数
  "ip": 0,             // I(强执行) 分数
  "next_dim": "BH",    // 下一维度
  "can_conclude": false, // 是否可得出结论
  "comment": "吐槽",    // ≤24字
  "scene": "场景描述",  // ≤80字
  "options": ["A","B","C","D"]  // 2-4个选项
}
📋 JSON 输出字段 (ASSESS) 
{
  "phase": "ASSESS",
  "q": <当前题号>,    // 递增
  "nb": <分数>,       // 偏N(能动)增 +1~3
  "bh": <分数>,       // 偏B(边界)增 +1~3
  "tf": <分数>,       // 偏T(理性)增 +1~3
  "ip": <分数>,       // 偏I(执行)增 +1~3
  "next_dim": "BH",    // NB/BH/TF/IP/END
  "can_conclude": false, // AI判断可结束时true
  "comment": "吐槽",    // ≤35字 必须有梗
  "scene": "场景描述",  // ≤80字 有画面感
  "options": ["A","B","C","D"]  // 2-4个 差异要大
}
